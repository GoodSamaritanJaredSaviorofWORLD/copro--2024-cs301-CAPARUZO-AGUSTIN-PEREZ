CREATE TABLE [dbo].[Stats] (
    [Id]               INT          IDENTITY (1, 1) NOT NULL,
    [Username]         VARCHAR (50) NOT NULL,
    [Gender]           VARCHAR (50) NOT NULL,
    [Hair]             VARCHAR (50) NOT NULL,
    [HairColor]        VARCHAR (50) NOT NULL,
    [EyeColor]         VARCHAR (50) NOT NULL,
    [FacialHair]       VARCHAR (50) NOT NULL,
    [FacialExpression] VARCHAR (50) NOT NULL,
    [TopClothing]      VARCHAR (50) NOT NULL,
    [Bottoms]          VARCHAR (50) NOT NULL,
    [Footwear]         VARCHAR (50) NOT NULL,
    [Headwear]         VARCHAR (50) NOT NULL,
    [Accessories]      VARCHAR (50) NOT NULL,
    [MeleeWeapon]      VARCHAR (50) NOT NULL,
    [RangedWeapon]     VARCHAR (50) NOT NULL,
    [Throwables]       VARCHAR (50) NOT NULL,
    [SpecialWeapon]    VARCHAR (50) NOT NULL,
    [Boosts]           VARCHAR (50) NOT NULL,
    [HealingItems]     VARCHAR (50) NOT NULL,
    [PetCompanion]     VARCHAR (50) NOT NULL,
    [Vehicle]          VARCHAR (50) NOT NULL,
    [Nightvision]      BIT          NOT NULL,
    [NoiseSensitivity] BIT          NOT NULL,
    [Skillpoints]      INT          NOT NULL,
    [Strength]         INT          NOT NULL,
    [Speed]            INT          NOT NULL,
    [Stl]              INT          NOT NULL,
    [defense]          INT          NOT NULL,
    PRIMARY KEY CLUSTERED ([Id] ASC)
);
